\section{Giới thiệu Tổng quan}
NoadDVo Geometry Studio là một ứng dụng web mạnh mẽ chuyên mô phỏng không gian hình học trực quan, hỗ trợ thiết kế, vẽ hình, phân tích toán học và tự động xuất mã nguồn LaTeX (TikZ). 

Được xây dựng với kiến trúc hiện đại, ứng dụng mang đến không gian Canvas vô hạn, khả năng bắt dính thông minh (Smart Snapping), cùng hệ thống quản lý giao diện đa dạng (Multi-Theme System) nổi bật với hai phong cách: \textbf{Neo-Brutalism} sáng tạo và \textbf{Tactical Dark} chuyên nghiệp, giúp người dùng tập trung tối đa vào việc thiết kế.

\section{Hướng dẫn Sử dụng Công cụ (Tool Usage)}
Dưới đây là danh sách chi tiết các công cụ (được phân loại theo nhóm) đang hiển thị trên thanh công cụ bên trái (Left Toolbar) cùng với tên tiếng Anh chuẩn xác và hướng dẫn cách thao tác từng bước.

\subsection{1. Nhóm Thao tác Cơ bản (Basic)}
\begin{itemize}[label=\textbullet]
    \item \textbf{Select (Chọn đối tượng):} Dùng để chọn một hoặc nhiều đối tượng. Click chuột vào một đối tượng để chọn. Có thể click và kéo thả để tạo vùng chọn (box selection). Giữ phím \texttt{Ctrl/Cmd} khi click để chọn thêm nhiều đối tượng cùng lúc.
    \item \textbf{Move (Di chuyển không gian/đối tượng):} Cho phép click và kéo một đối tượng hình học tự do (không bị ràng buộc) đến vị trí mới. Nếu click vào khoảng trống trên Canvas, bạn có thể kéo thả để di chuyển góc nhìn (Pan). \textit{Mẹo: Bạn có thể nhấn giữ phím Space ở bất kỳ công cụ nào để tạm thời kích hoạt tính năng Pan.}
\end{itemize}

\subsection{2. Nhóm Điểm (Points)}
\begin{itemize}[label=\textbullet]
    \item \textbf{Point (Điểm tự do):} Click vào một vị trí trống trên màn hình để tạo điểm tự do. Nếu click lên một đoạn thẳng, tia, hoặc đường tròn, điểm đó sẽ bị ràng buộc trượt trên đối tượng đó.
    \item \textbf{Midpoint (Trung điểm):} Dùng để xác định trung điểm. Bạn có thể click trực tiếp vào một đoạn thẳng (Segment) để lấy trung điểm của nó, hoặc click lần lượt vào hai điểm riêng biệt.
    \item \textbf{Intersection (Giao điểm):} Tìm điểm giao nhau giữa 2 đối tượng (đường thẳng, tia, đoạn thẳng, đường tròn). Chọn công cụ, sau đó click lần lượt vào đối tượng thứ nhất và đối tượng thứ hai. Giao điểm (nếu có) sẽ tự động xuất hiện.
\end{itemize}

\subsection{3. Nhóm Đường thẳng \& Đoạn thẳng (Lines \& Vectors)}
\begin{itemize}[label=\textbullet]
    \item \textbf{Segment (Đoạn thẳng):} Click điểm thứ nhất để chọn vị trí bắt đầu, click điểm thứ hai để kết thúc. Hệ thống sẽ vẽ đoạn thẳng nối hai điểm.
    \item \textbf{Line (Đường thẳng):} Cách dùng tương tự Segment, click 2 điểm, nhưng đường thẳng này sẽ kéo dài vô hạn về cả hai phía.
    \item \textbf{Ray (Tia):} Click điểm gốc trước, sau đó click điểm hướng đến. Tia sẽ kéo dài vô hạn từ điểm gốc đi qua điểm thứ 2.
    \item \textbf{Vector (Vectơ):} Click điểm gốc (điểm đặt) và điểm ngọn (điểm hướng). Mũi tên sẽ được tự động vẽ tại điểm ngọn để biểu diễn vectơ.
\end{itemize}

\subsection{4. Nhóm Tương quan Không gian (Constrained Lines)}
\begin{itemize}[label=\textbullet]
    \item \textbf{Perpendicular Line (Đường vuông góc):} Chọn công cụ, click vào một điểm muốn đường thẳng đi qua, sau đó click vào đoạn thẳng/đường thẳng tham chiếu. Kết quả tạo ra một đường vô hạn vuông góc với tham chiếu.
    \item \textbf{Parallel Line (Đường song song):} Chọn công cụ, click vào điểm muốn đi qua, sau đó click vào đoạn/đường thẳng tham chiếu. Một đường thẳng song song vô hạn sẽ được tạo ra.
\end{itemize}

\subsection{5. Nhóm Hình học Nâng cao \& Đường cong (Curves \& Shapes)}
\begin{itemize}[label=\textbullet]
    \item \textbf{Circle (Đường tròn):} Click điểm đầu tiên làm tâm đường tròn. Click điểm thứ hai để xác định một điểm nằm trên đường tròn (hoặc bán kính).
    \item \textbf{Ellipse (Elip):} Click chọn 2 tiêu điểm (focus) của Elip, sau đó di chuyển chuột và click điểm thứ 3 để xác định độ mở (một điểm nằm trên viền elip).
    \item \textbf{Hyperbola (Hypebol):} Tương tự Elip, click chọn 2 tiêu điểm, sau đó click điểm thứ 3 nằm trên đường hypebol để chốt độ cong.
    \item \textbf{Polynomial (Đa thức):} Click liên tiếp nhiều điểm để tạo đường cong trơn đi qua các điểm đó.
    \item \textbf{Polygon (Đa giác):} Nhấn chọn liên tiếp các điểm đóng vai trò là đỉnh của đa giác. Khi đã đủ, click lại vào điểm đầu tiên một lần nữa để khép kín hình.
\end{itemize}

\subsection{6. Nhóm Phép biến hình (Transformations)}
\begin{itemize}[label=\textbullet]
    \item \textbf{Reflect Point (Đối xứng qua điểm - Tâm đối xứng):} Chọn đối tượng (điểm/đường/hình) muốn lấy đối xứng, sau đó click vào một Điểm đóng vai trò là tâm đối xứng.
    \item \textbf{Reflect Line (Đối xứng qua đường thẳng - Trục đối xứng):} Chọn đối tượng muốn lấy đối xứng, sau đó click vào một Đường/Đoạn thẳng đóng vai trò là trục đối xứng.
    \item \textbf{Rotate Point (Phép quay):} (Tính năng đang phát triển/hoàn thiện).
    \item \textbf{Translate Vector (Tịnh tiến theo vectơ):} Chọn đối tượng muốn tịnh tiến, sau đó click vào một Vectơ tham chiếu đã có trên màn hình.
    \item \textbf{Dilate Point (Vị tự):} Lấy vị tự của một đối tượng qua một tâm với một hệ số (hiện đang được mở rộng thông qua giao diện thuộc tính).
\end{itemize}

\subsection{7. Nhóm Đo lường \& Chú thích (Measurements \& Annotations)}
\begin{itemize}[label=\textbullet]
    \item \textbf{Angle (Đo Góc):} Click lần lượt 3 điểm theo thứ tự, trong đó điểm thứ 2 (điểm giữa) là đỉnh của góc. Công cụ sẽ vẽ cung tròn đánh dấu và ghi số đo góc.
    \item \textbf{Distance (Đo khoảng cách):} Click 2 điểm để hệ thống tính toán và hiển thị khoảng cách tuyệt đối giữa chúng.
    \item \textbf{Area (Diện tích):} Click chọn một vùng khép kín hoặc đa giác để hiển thị diện tích của nó.
    \item \textbf{Text (Văn bản):} Click vào một vị trí trên Canvas. Một hộp thoại sẽ hiện lên để bạn nhập nội dung (hỗ trợ nhập cả công thức toán học).
\end{itemize}

\subsection{8. Nhóm Công cụ Đặc biệt (Special Tools)}
\begin{itemize}[label=\textbullet]
    \item \textbf{Trim (Cắt xén):} Đây là công cụ rất mạnh mẽ. Khi các đường thẳng/cung tròn giao nhau tạo thành nhiều mảnh, bạn hãy chọn Trim và click vào những phần viền thừa. Phần được click sẽ bị xóa đi đến các giao điểm gần nhất.
    \item \textbf{Fill (Đổ màu):} Click vào một vùng trống nằm giữa các ranh giới khép kín (được tạo bởi sự giao nhau của nhiều đoạn thẳng/đường tròn). Vùng đó sẽ được tự động nhận diện thành một Region (khu vực) và đổ màu nền.
    \item \textbf{Slider (Thanh trượt):} Click vào Canvas để tạo ra một biến số (variable) có thể thay đổi liên tục, dùng để gán cho các tham số toán học động.
\end{itemize}

\section{Hệ thống Quản lý (Workspace Panels)}

\subsection{Cây Đối tượng (Object Tree)}
Nằm ở thanh Sidebar trái, cho phép quản lý cấu trúc phân cấp các đối tượng:
\begin{itemize}[label=\textbullet]
    \item Dễ dàng đổi tên đối tượng bằng cách click đúp (Double-click) vào nhãn tên.
    \item Nhấn icon \textbf{Mắt} để ẩn/hiện đối tượng.
    \item Nhấn icon \textbf{Khóa} để khóa không cho chỉnh sửa trực tiếp trên Canvas.
    \item Kiểm tra quan hệ phụ thuộc (\textbf{Dependencies}) giữa các đối tượng ở cuối bảng để biết đối tượng này sinh ra từ đâu.
\end{itemize}

\subsection{Bảng Thuộc tính (Properties Inspector)}
Nằm ở thanh Sidebar phải, tự động hiển thị thông số của đối tượng đang được chọn:
\begin{itemize}[label=\textbullet]
    \item Thay đổi màu viền, màu nền (Opacity), độ dày nét vẽ (Stroke Width), kiểu nét đứt/liền.
    \item Bật/tắt hiển thị nhãn tên (Label) và điều chỉnh phông chữ.
    \item \textbf{Tính năng Khóa (Lock Panel):} Nhấn vào icon ổ khóa ở góc phải trên cùng của Panel để ghim chặt bảng thuộc tính cho đối tượng hiện tại. Khi khóa, bạn có thể click chọn đối tượng khác (hoặc dùng Fill/Trim) mà Properties Inspector không bị nhảy sang đối tượng mới, rất hữu ích khi bạn muốn tinh chỉnh thông số cho một vùng đang thao tác.
\end{itemize}

\section{Xuất file \& Tích hợp (Export \& Integrations)}
\begin{itemize}[label=\textbullet]
    \item \textbf{TikZ Panel:} Hệ thống thời gian thực tự động chuyển đổi các đối tượng hình học sang mã TikZ của LaTeX. Bạn chỉ cần mở TikZ Panel trên thanh menu, copy mã và dán vào trình soạn thảo LaTeX (như Overleaf, TeXStudio).
    \item \textbf{Command Palette:} Nhấn \texttt{Ctrl+K} (hoặc \texttt{Cmd+K}) để mở bảng lệnh. Gõ tên công cụ hoặc lệnh để thực thi nhanh chóng không cần dùng chuột.
\end{itemize}
