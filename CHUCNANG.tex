\section{Giới thiệu Tổng quan}
\textbf{NoadDVo Untitled Geometry} là một phần mềm kiến thiết hình học trực tuyến chuyên nghiệp. Với phong cách thiết kế \textbf{Neo-Brutalism} hiện đại (viền đen dày, đổ bóng cứng cáp), phần mềm giúp giáo viên, học sinh và sinh viên dễ dàng dựng các cấu trúc hình học phức tạp. Đặc biệt, hệ thống hoạt động như một \textit{trình biên dịch hình học} – tự động sinh mã nguồn \LaTeX (gói \texttt{TikZ}) siêu chuẩn xác ngay trong thời gian thực. Mọi đối tượng đều dựa trên phương trình Vector và tính toán tọa độ nghiêm ngặt.

\section{Hệ thống 7 Nhóm Công Cụ Hình Học (Left Toolbar)}
Để sử dụng phần mềm hiệu quả, bạn cần nắm rõ 7 nhóm công cụ chính nằm ở thanh Toolbar bên trái. Một số công cụ được đặt trong Menu thả xuống (Dropdown); bạn chỉ cần click vào biểu tượng đại diện để mở toàn bộ công cụ ẩn bên trong.

\subsection{1. Nhóm Công cụ Chuột (Action Group)}
Nhóm này bao gồm các chức năng tương tác cơ bản nhất với đối tượng.
\begin{itemize}[label=\textbullet]
    \item \textbf{Select (Mũi tên Chọn):}
    \begin{itemize}
        \item \textit{Chức năng:} Chọn đối tượng để xem và thay đổi thuộc tính.
        \item \textit{Thao tác:} Trỏ chuột và click trực tiếp vào đường viền của đối tượng (hoặc click vào phần diện tích bên trong nếu đối tượng đã được tô màu/Fill). Có thể quét khối (Drag) để chọn nhiều đối tượng cùng lúc. Sau khi chọn, bảng Properties bên phải sẽ hiện ra để bạn đổi màu nét (Stroke), màu nền (Fill), độ dày nét (Width), đổi kiểu nét đứt, hoặc ẩn/hiện nhãn (Label).
    \end{itemize}
    \item \textbf{Move (Di chuyển Bàn tay):}
    \begin{itemize}
        \item \textit{Chức năng:} Di dời vị trí của một "Điểm tự do".
        \item \textit{Thao tác:} Click và giữ chuột trái vào một Điểm tự do (Free Point) trên Canvas, sau đó kéo đến tọa độ mới. Toàn bộ các đối tượng hình học phụ thuộc vào điểm đó sẽ tự động tính toán lại phương trình và chuyển động theo.
    \end{itemize}
\end{itemize}

\subsection{2. Nhóm Điểm (Point Group)}
Nhóm công cụ kiến tạo các cấu phần cơ bản nhất của hình học: Điểm.
\begin{itemize}[label=\textbullet]
    \item \textbf{Point (Tạo Điểm):}
    \begin{itemize}
        \item \textit{Chức năng:} Khởi tạo một điểm mới.
        \item \textit{Thao tác:} 
        \begin{itemize}
            \item Click vào khoảng trống bất kỳ trên Grid để tạo \textbf{Điểm tự do} (có thể di chuyển được).
            \item Click lên một cạnh, một đường thẳng hoặc một đường tròn để tạo \textbf{Điểm thuộc đường} (chỉ có thể trượt dọc theo đường đó).
        \end{itemize}
    \end{itemize}
    \item \textbf{Midpoint (Trung điểm):}
    \begin{itemize}
        \item \textit{Chức năng:} Lấy điểm chính giữa của một đoạn thẳng.
        \item \textit{Thao tác:} Click tuần tự vào \textbf{2 Điểm} bất kỳ, hoặc đơn giản hơn là click trực tiếp vào \textbf{1 Đoạn thẳng} (Segment) đã có sẵn.
    \end{itemize}
    \item \textbf{Intersection (Giao điểm):}
    \begin{itemize}
        \item \textit{Chức năng:} Xác định tất cả các điểm giao cắt giữa hai đường.
        \item \textit{Thao tác:} Click lần lượt vào \textbf{Đối tượng thứ nhất} (Ví dụ: Đường thẳng) $\rightarrow$ Click vào \textbf{Đối tượng thứ hai} (Ví dụ: Đường tròn). Tất cả các giao điểm sẽ tự động xuất hiện.
    \end{itemize}
\end{itemize}

\subsection{3. Nhóm Đường thẳng (Line Group)}
Cung cấp các công cụ nối điểm và dựng đường tương quan định hướng.
\begin{itemize}[label=\textbullet]
    \item \textbf{Line (Đường thẳng):} Đường thẳng vô tận. Thao tác: Click chọn 2 điểm phân biệt.
    \item \textbf{Segment (Đoạn thẳng):} Giới hạn giữa 2 điểm. Thao tác: Click Điểm đầu $\rightarrow$ Click Điểm cuối.
    \item \textbf{Ray (Tia):} Kéo dài vô tận về một phía. Thao tác: Click Điểm gốc (Start) $\rightarrow$ Click Điểm đi qua (Through).
    \item \textbf{Vector (Vectơ):} Có hướng mũi tên. Thao tác: Click Điểm gốc $\rightarrow$ Click Điểm ngọn (Đầu mũi tên).
    \item \textbf{Parallel (Đường song song):} 
    \begin{itemize}
        \item \textit{Chức năng:} Dựng đường thẳng đi qua 1 điểm và song song với 1 đường thẳng khác.
        \item \textit{Thao tác:} Click chọn \textbf{Điểm đi qua} $\rightarrow$ Click chọn \textbf{Đường thẳng làm chuẩn}.
    \end{itemize}
    \item \textbf{Perpendicular (Đường vuông góc):}
    \begin{itemize}
        \item \textit{Chức năng:} Dựng đường thẳng đi qua 1 điểm và vuông góc với 1 đường thẳng khác.
        \item \textit{Thao tác:} Click chọn \textbf{Điểm đi qua} $\rightarrow$ Click chọn \textbf{Đường thẳng làm chuẩn}.
    \end{itemize}
\end{itemize}

\subsection{4. Nhóm Đường tròn \& Đa giác (Shape Group)}
Khởi tạo các đối tượng hình khép kín và các đường cong cung.
\begin{itemize}[label=\textbullet]
    \item \textbf{Circle (Đường tròn):} Thao tác bằng 2 lần click: Click chọn \textbf{Tâm} $\rightarrow$ Click chọn \textbf{Điểm đi qua} (xác định bán kính).
    \item \textbf{Arc (Cung tròn):} Dựng cung chuẩn xác qua 3 lần click:
    \begin{enumerate}
        \item Click chọn \textbf{Tâm}.
        \item Click chọn \textbf{Điểm đầu} (Thao tác này đồng thời xác định bán kính của cung và Góc xuất phát).
        \item Click chọn \textbf{Điểm cuối} (Xác định Góc kết thúc).
        \item \textit{Lưu ý:} Cung luôn được quét theo chiều ngược chiều kim đồng hồ (Counter-Clockwise) từ Điểm đầu đến Điểm cuối.
    \end{enumerate}
    \item \textbf{Elliptical Arc (Cung Elíp):} Hữu ích để vẽ thiết diện hình trụ, nón, hình học không gian. Thao tác qua 3 lần click:
    \begin{enumerate}
        \item Click chọn \textbf{Tâm}.
        \item Click chọn \textbf{Điểm xác định trục ngang} (Định hình bán kính $R_x$ lớn và Góc xuất phát).
        \item Click chọn \textbf{Điểm xác định Góc kết thúc}.
        \item \textit{Lưu ý:} Bán kính trục bé $R_y$ được tự động tính bằng $0.6 R_x$. Bạn có thể vào bảng \textbf{Properties} $\rightarrow$ điều chỉnh trường số \textbf{"y radius"} để tự do làm dẹt/mở rộng độ cong của cung.
    \end{enumerate}
    \item \textbf{Circumcircle (Đường tròn ngoại tiếp):} Thao tác: Click tuần tự vào \textbf{3 Điểm phân biệt} không thẳng hàng.
    \item \textbf{Incircle (Đường tròn nội tiếp):} Thao tác: Click tuần tự vào \textbf{3 Điểm} tạo thành 1 tam giác.
    \item \textbf{Polygon (Đa giác):} 
    \begin{itemize}
        \item \textit{Thao tác:} Click chuột liên tiếp lên không gian lưới để chấm các đỉnh của đa giác.
        \item \textit{Hoàn thành:} Để kết thúc khối hình, hãy \textbf{click lại vào Điểm (Đỉnh) đầu tiên} mà bạn đã tạo.
    \end{itemize}
\end{itemize}

\subsection{5. Nhóm Đường Cô-nic (Conic Group)}
Nhóm công cụ phục vụ đại số bậc cao.
\begin{itemize}[label=\textbullet]
    \item \textbf{Ellipse (Hình Elíp):} Thao tác: Click \textbf{Tiêu điểm thứ 1} $\rightarrow$ Click \textbf{Tiêu điểm thứ 2} $\rightarrow$ Click \textbf{1 Điểm nằm trên Elíp}.
    \item \textbf{Hyperbola:} Thao tác: Click \textbf{Tiêu điểm thứ 1} $\rightarrow$ Click \textbf{Tiêu điểm thứ 2} $\rightarrow$ Click \textbf{1 Điểm nằm trên Hyperbola}.
    \item \textbf{Polynomial (Hàm Đa thức):} Thao tác: Click liên tục các điểm tập mẫu. Đường cong trơn nội suy theo phương trình đa thức sẽ chạy qua tất cả các điểm đó.
\end{itemize}

\subsection{6. Nhóm Phép Biến Hình (Transform Group)}
Tự động sao chép và nhân bản đối tượng dựa trên các phép biến đổi toán học.
\begin{itemize}[label=\textbullet]
    \item \textbf{Reflect (Phép Đối xứng):} 
    \begin{itemize}
        \item \textit{Chức năng:} Lấy đối xứng trục hoặc đối xứng tâm.
        \item \textit{Thao tác:} Click \textbf{Đối tượng} cần lấy đối xứng $\rightarrow$ Click \textbf{Đường thẳng} làm Trục đối xứng (hoặc click \textbf{Điểm} làm Tâm đối xứng).
    \end{itemize}
    \item \textbf{Rotate (Phép Quay):}
    \begin{itemize}
        \item \textit{Thao tác:} Click \textbf{Đối tượng} cần quay $\rightarrow$ Click \textbf{Điểm} làm Tâm quay. Hộp thoại nhập Góc sẽ hiện lên $\rightarrow$ Nhập số độ (VD: 90, 45).
    \end{itemize}
    \item \textbf{Translate (Phép Tịnh tiến):}
    \begin{itemize}
        \item \textit{Thao tác:} Click \textbf{Đối tượng} cần tịnh tiến $\rightarrow$ Click \textbf{1 Vector} đã có sẵn (hoặc click 2 điểm liên tiếp để xác định hướng tịnh tiến).
    \end{itemize}
    \item \textbf{Dilate (Phép Vị tự):}
    \begin{itemize}
        \item \textit{Thao tác:} Click \textbf{Đối tượng} $\rightarrow$ Click \textbf{Điểm} làm Tâm vị tự $\rightarrow$ Nhập \textbf{Tỷ số vị tự $k$} vào hộp thoại. (Ví dụ: 0.5 để thu nhỏ một nửa).
    \end{itemize}
\end{itemize}

\subsection{7. Nhóm Đo Lường \& Tiện Ích (Measure \& Utils Group)}
Cung cấp các công cụ bổ trợ, gắn nhãn số liệu và trang trí hình.
\begin{itemize}[label=\textbullet]
    \item \textbf{Angle (Đo Góc):} 
    \begin{itemize}
        \item \textit{Thao tác:} Click tuần tự \textbf{3 Điểm}. Đặc biệt quan trọng: Điểm click thứ 2 bắt buộc phải là \textbf{Đỉnh của góc}. Ví dụ muốn đo góc $\angle ABC$, hãy click $A \rightarrow B \rightarrow C$.
    \end{itemize}
    \item \textbf{Distance (Đo Khoảng cách / Độ dài):}
    \begin{itemize}
        \item \textit{Thao tác:} Click vào \textbf{2 Điểm} (tính khoảng cách 2 điểm), hoặc click \textbf{1 Điểm và 1 Đường thẳng} (tính khoảng cách từ điểm đến đường). Thông số nhãn sẽ tự hiện ra.
    \end{itemize}
    \item \textbf{Area (Tính Diện tích):} Thao tác: Click trực tiếp vào 1 \textbf{Đa giác} (Polygon) hoặc 1 \textbf{Đường tròn} khép kín.
    \item \textbf{Text (Nhập Văn bản):} 
    \begin{itemize}
        \item \textit{Thao tác:} Click vào bất kỳ vùng trống nào trên Grid. 
        \item \textit{Tính năng:} Bạn có thể gõ chữ cái thông thường hoặc nhập công thức toán học $\LaTeX$ kẹp giữa dấu \$. Ví dụ: \verb|$x = \frac{-b \pm \sqrt{\Delta}}{2a}$|.
    \end{itemize}
    \item \textbf{Fill (Tô màu diện tích khép kín):} 
    \begin{itemize}
        \item \textit{Chức năng:} Tô màu phần giới hạn bởi các đoạn thẳng đan chéo nhau.
        \item \textit{Thao tác:} Click vào một mảng diện tích kín trên Grid. Hệ thống sẽ quét thuật toán Scanline, tự động bọc 1 đa giác ảo vào trong và đổ màu vào vùng diện tích đó.
    \end{itemize}
    \item \textbf{Trim (Cắt bỏ nét):}
    \begin{itemize}
        \item \textit{Chức năng:} Xóa phần đường bị dư thừa, chia cắt bởi các đường khác.
        \item \textit{Thao tác:} Click trực tiếp lên đoạn đường dư để cắt bỏ.
    \end{itemize}
\end{itemize}
