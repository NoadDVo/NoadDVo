\section{Giới thiệu Tổng quan}
\textbf{NoadDVo Untitled Geometry} là một ứng dụng thiết kế và mô phỏng hình học trực quan, nổi bật với phong cách thiết kế \textbf{Neo-Brutalism} (viền đen dày, đổ bóng cứng, màu nhấn cam đặc trưng). Ứng dụng cung cấp một không gian làm việc chuyên nghiệp để vẽ hình toán học và tự động sinh mã nguồn LaTeX (TikZ).

\section{Cấu trúc Giao diện Chính (Workspace Layout)}
Giao diện của ứng dụng được bố trí thông minh thành 5 khu vực chính:

\subsection{1. Thanh Điều Hướng (Top Bar)}
Nằm ở trên cùng, chứa các nút chức năng tổng thể:
\begin{itemize}[label=\textbullet]
    \item \textbf{Logo \& Tên dự án:} NOADDVO UNTITLED GEOMETRY.
    \item \textbf{Undo / Redo:} Hai nút mũi tên quay lại/tiến tới giúp hoàn tác các thao tác vẽ.
    \item \textbf{Menu Công cụ (Góc phải):} Bao gồm \textbf{PROJECT} (quản lý file dự án), \textbf{EXPORT} (xuất hình ảnh/mã), \textbf{THEME} (đổi giao diện), và nút \textbf{Help (?)}.
\end{itemize}

\subsection{2. Bảng Vẽ Trung Tâm (Main Canvas)}
Khu vực lưới (Grid) vô hạn chiếm phần lớn diện tích màn hình.
\begin{itemize}[label=\textbullet]
    \item Khi chưa có đối tượng nào, bảng vẽ hiển thị hộp thoại \textbf{EMPTY WORKSPACE} với thông điệp "CREATE YOUR FIRST OBJECT" cùng 2 nút hành động nhanh: \textbf{CREATE A POINT} (Tạo điểm đầu tiên) và \textbf{LOAD EXAMPLE} (Tải một hình mẫu có sẵn).
    \item Bạn có thể dùng chuột để vẽ, kéo thả, hoặc giữ phím Space để Pan (di chuyển) góc nhìn.
\end{itemize}

\subsection{3. Thanh Công Cụ (Left Toolbar)}
Thanh công cụ nổi nằm bên trái màn hình, chứa các nhóm công cụ hình học (có menu xổ ra khi click):
\begin{itemize}[label=\textbullet]
    \item \textbf{Action (Mũi tên chuột):} Chứa công cụ Select (chọn) và Move (di chuyển).
    \item \textbf{Point (Điểm):} Khởi tạo Point, Midpoint, Intersection.
    \item \textbf{Line (Đường chéo):} Vẽ Line, Segment, Ray, Vector, và các đường tương quan (Parallel, Perpendicular...).
    \item \textbf{Shape (Đường tròn):} Vẽ Circle, Circumcircle, Incircle, Polygon.
    \item \textbf{Conic (Quỹ đạo):} Khởi tạo Ellipse, Hyperbola, Polynomial.
    \item \textbf{Transform (Phép biến hình):} Reflect, Rotate, Translate, Dilate.
    \item \textbf{Measure \& Utils (Đồng hồ đo):} Angle, Distance, Area, Text, Trim, Fill, Slider.
\end{itemize}

\subsection{4. Thanh Quản Lý (Right Sidebar)}
Nằm ở bên phải màn hình, chia làm 2 panel xếp chồng lên nhau:
\begin{itemize}[label=\textbullet]
    \item \textbf{WORKSPACE / OBJECT TREE:}
    \begin{itemize}
        \item Cung cấp thanh tìm kiếm (\textit{Search objects}).
        \item Hệ thống bộ lọc (Filters) phân loại đối tượng: \textbf{ALL, POINTS, LINES, CIRCLES, CONST}...
        \item Hiển thị danh sách toàn bộ các cấu phần hình học và xem xét mối liên hệ phụ thuộc (\textit{Dependency links}) giữa chúng.
    \end{itemize}
    \item \textbf{INSPECTOR / PROPERTIES:}
    \begin{itemize}
        \item Hiển thị thuộc tính của đối tượng đang được chọn (màu sắc, độ dày nét, hiển thị nhãn...).
        \item Cung cấp chức năng \textbf{Lock Panel} (Biểu tượng ổ khóa ở tiêu đề): Giúp ghim chặt bảng thuộc tính cho đối tượng hiện hành, không bị mất tiêu điểm khi click ra ngoài.
    \end{itemize}
\end{itemize}

\subsection{5. Khung Xuất Mã (Bottom Panel) \& Thanh Trạng Thái (Status Bar)}
\begin{itemize}[label=\textbullet]
    \item \textbf{GENERATED TIKZ:} Khung hiển thị nằm ngang ở dưới cùng, tự động sinh mã nguồn LaTeX/TikZ (VD: \texttt{\textbackslash begin\{tikzpicture\}...}). Hỗ trợ tùy chỉnh \textit{FORM: ACADEMIC, WRAP, READONLY} và nút \textbf{COPY} nhanh.
    \item \textbf{Status Bar:} Nằm sát mép dưới cùng, hiển thị các thông số thời gian thực: \textit{ZOOM 100\%, Tọa độ X/Y của chuột, Số lượng đối tượng đang chọn (SELECTION), Chế độ SNAP (Grid 1), MODE, và trạng thái TIKZ.}
\end{itemize}

\section{Hướng dẫn Thao tác Nhanh}
\begin{enumerate}
    \item \textbf{Tạo đối tượng:} Bắt đầu bằng cách chọn công cụ \textit{Line} (như trên ảnh đang được chọn màu cam), click 2 điểm trên Canvas để vẽ một đoạn thẳng.
    \item \textbf{Kiểm soát thuộc tính:} Khi đoạn thẳng được chọn, \textit{INSPECTOR} sẽ hiện lên thông số để bạn đổi nét liền thành nét đứt.
    \item \textbf{Lấy mã LaTeX:} Toàn bộ thay đổi sẽ được phản ánh ngay lập tức ở khung \textit{TIKZ} bên dưới. Nhấn nút Copy và dán vào phần mềm LaTeX của bạn.
\end{enumerate}
